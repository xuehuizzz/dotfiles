### 安装
```bash
brew install neovim
mkdir -p ~/.config/nvim
touch ~/.config/nvim/init.lua

# 安装 vim-plug 插件管理器
sh -c 'curl -fLo "${XDG_DATA_HOME:-$HOME/.local/share}"/nvim/site/autoload/plug.vim --create-dirs \
       https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim'
```
